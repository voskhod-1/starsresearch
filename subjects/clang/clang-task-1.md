### Все, что закомментировано – придирки
```cpp
#include <iostream>
#include <cmath>

int main() {
    double sum = 0.0;
    double f, t = 1; 
    int n = 1; 
    //1/2+ 1/4 *2 + 1/8* 3 
    do {
        //f = 1.0 / (pow(2, n) * n);
        t *= 1.0 / 2.0 ;
        //f = t * (1.0 / n);
        sum += f = t * (1.0 / n++);
        //n++;
    } while (f > 1e-12);

    std::cout << "сумма= " << sum << std::endl;
    std::cout << 1.0 / n << std::endl;
    std::cout << "0,000001 " << 1e-6 << std::endl;
    std::cout << "ln(2): " << log(2) << std::endl;
    return 0;
}
```
